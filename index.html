const tg = window.Telegram?.WebApp;

// ── Canvas ──
const canvas = document.getElementById('game');
const ctx = canvas.getContext('2d');
const starCanvas = document.getElementById('canvas-stars');
const starCtx = starCanvas.getContext('2d');

function resize() {
  canvas.width = starCanvas.width = window.innerWidth;
  canvas.height = starCanvas.height = window.innerHeight;
}
resize();
window.addEventListener('resize', () => { resize(); initStars(); });

// ── Stars + Grid ──
let stars = [], gridAnim = 0;
function initStars() {
  stars = Array.from({length: 100}, () => ({
    x: Math.random() * canvas.width,
    y: Math.random() * canvas.height,
    r: Math.random() * 1.4 + 0.2,
    speed: Math.random() * 0.4 + 0.1,
    alpha: Math.random() * 0.6 + 0.2
  }));
}
initStars();

function drawBackground() {
  starCtx.clearRect(0, 0, canvas.width, canvas.height);
  gridAnim += 0.003;

  // Grid lines (blockchain grid effect)
  const gSize = 60;
  const offset = (gridAnim * gSize * 0.5) % gSize;
  starCtx.strokeStyle = 'rgba(124,58,237,0.07)';
  starCtx.lineWidth = 0.5;
  for (let x = -gSize + (offset % gSize); x < canvas.width + gSize; x += gSize) {
    starCtx.beginPath(); starCtx.moveTo(x, 0); starCtx.lineTo(x, canvas.height); starCtx.stroke();
  }
  for (let y = -gSize + (offset % gSize); y < canvas.height + gSize; y += gSize) {
    starCtx.beginPath(); starCtx.moveTo(0, y); starCtx.lineTo(canvas.width, y); starCtx.stroke();
  }

  // Stars
  stars.forEach(s => {
    s.y += s.speed;
    if (s.y > canvas.height) { s.y = 0; s.x = Math.random() * canvas.width; }
    starCtx.beginPath();
    starCtx.arc(s.x, s.y, s.r, 0, Math.PI * 2);
    starCtx.fillStyle = `rgba(200,180,255,${s.alpha})`;
    starCtx.fill();
  });
}

  ctx.translate(p.x, p.y);

    ctx.restore(); return;
  }

  // Thruster flames
  p.thruster += 0.18;
  const fl = 16 + Math.sin(p.thruster) * 9;
  const fLeft = 12 + Math.sin(p.thruster + 1) * 5;

  // Left engine

  g1.addColorStop(0, 'rgba(139,92,246,0.9)');
  g1.addColorStop(1, 'rgba(139,92,246,0)');
  ctx.beginPath(); ctx.moveTo(-18, 6); ctx.lineTo(-12, 6+fLeft); ctx.lineTo(-6, 6);
  ctx.fillStyle = g1; ctx.fill();

  // Right engine
  const g2 = ctx.createLinearGradient(12, 6, 12, 6 + fLeft);
  g2.addColorStop(0, 'rgba(139,92,246,0.9)');
  g2.addColorStop(1, 'rgba(139,92,246,0)');
  ctx.beginPath(); ctx.moveTo(6, 6); ctx.lineTo(12, 6+fLeft); ctx.lineTo(18, 6);
  ctx.fillStyle = g2; ctx.fill();

  // Center booster
  const g3 = ctx.createLinearGradient(0, 10, 0, 10+fl);
  g3.addColorStop(0, 'rgba(6,182,212,1)');
  g3.addColorStop(0.5, 'rgba(124,58,237,0.7)');
  g3.addColorStop(1, 'rgba(6,182,212,0)');
  ctx.beginPath(); ctx.moveTo(-7, 8); ctx.lineTo(0, 10+fl); ctx.lineTo(7, 8);
  ctx.fillStyle = g3; ctx.fill();

  // Main body
  ctx.shadowBlur = 16; ctx.shadowColor = '#7c3aed';
  ctx.beginPath();
  ctx.moveTo(0, -PH/2);
  ctx.lineTo(PW/2 - 4, PH/2 - 8);

  bodyGrad.addColorStop(0.5, '#7c3aed');
  bodyGrad.addColorStop(1, '#4c1d95');
  ctx.fillStyle = bodyGrad; ctx.fill();

  ctx.shadowBlur = 0;

  ctx.shadowBlur = 12; ctx.shadowColor = '#06b6d4';
  ctx.beginPath();
  ctx.ellipse(0, -PH/4, 8, 11, 0, 0, Math.PI*2);
  const ckGrad = ctx.createRadialGradient(-2, -PH/4-3, 1, 0, -PH/4, 8);
  ckGrad.addColorStop(0, '#e0f2fe');
  ckGrad.addColorStop(0.6, '#06b6d4');
  ckGrad.addColorStop(1, '#0e7490');
  ctx.fillStyle = ckGrad; ctx.fill();

  // Wings
  ctx.shadowBlur = 8; ctx.shadowColor = '#7c3aed';
  [[1],[−1]].forEach(([dir]) => {
    ctx.beginPath();
    ctx.moveTo(dir * (PW/2-4), PH/2-8);
    ctx.lineTo(dir * (PW/2+12), PH/2+4);
    ctx.lineTo(dir * (PW/2-2), PH/2-2);
    ctx.closePath();
    ctx.fillStyle = '#5b21b6'; ctx.fill();
  });

  // Shield
  if (p.shield) {
    ctx.shadowBlur = 18; ctx.shadowColor = '#06b6d4';
    ctx.beginPath();
    ctx.arc(0, 0, 34, 0, Math.PI*2);
    ctx.strokeStyle = `rgba(6,182,212,${0.4 + 0.3*Math.sin(frame*0.12)})`;
    ctx.lineWidth = 3; ctx.stroke();
    // Hex pattern
    ctx.strokeStyle = `rgba(6,182,212,${0.15 + 0.1*Math.sin(frame*0.12)})`;
    ctx.lineWidth = 1;
    for (let i = 0; i < 6; i++) {
      const a = i * Math.PI/3 + frame * 0.01;
      ctx.beginPath(); ctx.moveTo(0,0); ctx.lineTo(Math.cos(a)*34, Math.sin(a)*34); ctx.stroke();
    }
  }

  ctx.restore();
}

// Enemy configs
const ENEMY_TYPES = [
  { name:'Hacker',    color:'#ef4444', glow:'#ef4444', score:10, shape:'hexagon',  hp:1 },
  { name:'Scammer',   color:'#f97316', glow:'#f97316', score:15, shape:'diamond',  hp:1 },
  { name:'Bug',       color:'#8b5cf6', glow:'#8b5cf6', score:20, shape:'triangle', hp:1 },
  { name:'Dark Node', color:'#dc2626', glow:'#ff0000', score:50, shape:'boss',     hp:6 },
];

function drawEnemy(e) {
  ctx.save();
  ctx.translate(e.x, e.y);
  ctx.rotate(e.rot);

  const t = e.config;
  ctx.shadowBlur = 14; ctx.shadowColor = t.glow;

  if (t.shape === 'hexagon') {
    ctx.beginPath();
    for (let i = 0; i < 6; i++) {
      const a = i*Math.PI/3 - Math.PI/6;
      i===0 ? ctx.moveTo(Math.cos(a)*18, Math.sin(a)*18) : ctx.lineTo(Math.cos(a)*18, Math.sin(a)*18);
    }
    ctx.closePath(); ctx.fillStyle = t.color; ctx.fill();
    ctx.font = 'bold 11px monospace'; ctx.fillStyle='#fff';
    ctx.textAlign='center'; ctx.textBaseline='middle';
    ctx.fillText('☠', 0, 0);
  } else if (t.shape === 'diamond') {
    ctx.beginPath();
    ctx.moveTo(0,-20); ctx.lineTo(16,0); ctx.lineTo(0,20); ctx.lineTo(-16,0);
    ctx.closePath(); ctx.fillStyle = t.color; ctx.fill();
    ctx.font = '11px monospace'; ctx.fillStyle='#fff';
    ctx.textAlign='center'; ctx.textBaseline='middle';
    ctx.fillText('💸', 0, 0);
  } else if (t.shape === 'triangle') {
    ctx.beginPath();
    ctx.moveTo(0,-20); ctx.lineTo(18,18); ctx.lineTo(-18,18);
    ctx.closePath(); ctx.fillStyle = t.color; ctx.fill();
    ctx.font = '11px monospace'; ctx.fillStyle='#fff';
    ctx.textAlign='center'; ctx.textBaseline='middle';
    ctx.fillText('🐛', 0, 4);
  } else if (t.shape === 'boss') {
    // Dark Node boss - bigger
    ctx.shadowBlur = 24;
    ctx.beginPath();
    ctx.moveTo(0,-30); ctx.lineTo(26,-12); ctx.lineTo(26,12);
    ctx.lineTo(0,30); ctx.lineTo(-26,12); ctx.lineTo(-26,-12);
    ctx.closePath();
    const bGrad = ctx.createRadialGradient(0,0,4, 0,0,30);
    bGrad.addColorStop(0, '#7f1d1d'); bGrad.addColorStop(1, '#dc2626');
    ctx.fillStyle = bGrad; ctx.fill();
    // inner glow ring
    ctx.strokeStyle = `rgba(255,50,50,${0.5+0.4*Math.sin(frame*0.1)})`;
    ctx.lineWidth = 2; ctx.stroke();
    ctx.font = '16px monospace'; ctx.fillStyle='#fff';
    ctx.textAlign='center'; ctx.textBaseline='middle';
    ctx.fillText('💀', 0, 0);
    // HP bar
    const bw = 52;
    ctx.fillStyle = '#3f0000'; ctx.fillRect(-bw/2, 36, bw, 6);
    ctx.fillStyle = '#ef4444'; ctx.fillRect(-bw/2, 36, bw*(e.hp/e.maxHp), 6);
    // Name tag
    ctx.font = 'bold 8px Orbitron, monospace';
    ctx.fillStyle = '#fca5a5'; ctx.textAlign='center'; ctx.textBaseline='top';
    ctx.fillText('DARK NODE', 0, 44);
  }

  ctx.restore();
}

function spawnParticles(x, y, color, count=12, big=false) {
  for (let i = 0; i < count; i++) {
    const a = (Math.PI*2/count)*i + Math.random()*0.3;
    const spd = big ? Math.random()*5+2 : Math.random()*3+1;
    particles.push({
      x, y,
      vx: Math.cos(a)*spd, vy: Math.sin(a)*spd,
      life: 1, decay: big ? Math.random()*0.025+0.015 : Math.random()*0.04+0.02,
      r: big ? Math.random()*5+3 : Math.random()*3+1.5, color
    });
  }
}

function spawnTokens(x, y) {
  // Floating $DLI coins
  for (let i = 0; i < 3; i++) {
    tokens.push({
      x: x + (Math.random()-0.5)*30,
      y, vy: -2 - Math.random(),
      vx: (Math.random()-0.5)*1.5,
      life: 1, decay: 0.012, anim: Math.random()*Math.PI*2
    });
  }
}

function spawnPowerup(x, y) {
  if (Math.random() > 0.28) return;
  const types = [
    { id:'shield', icon:'🛡️', label:'SMART CONTRACT', color:'#06b6d4' },
    { id:'rapid',  icon:'⚡', label:'FLASH LOAN',    color:'#f59e0b' },
    { id:'bomb',   icon:'💣', label:'DAO BOMB',       color:'#ef4444' },
  ];
  powerups.push({
    x, y, vy: 1.4, life: 1, decay: 0.003,
    anim: 0, type: types[Math.floor(Math.random()*3)]
  });
}

function spawnEnemy() {
  const margin = 30;
  const x = margin + Math.random()*(canvas.width - margin*2);
  const pool = level < 3 ? [0,1] : level < 5 ? [0,1,2] : [0,1,2,3];
  // Boss less frequent
  const filteredPool = pool.filter(i => ENEMY_TYPES[i].shape !== 'boss' || Math.random() < 0.15);
  const idx = filteredPool[Math.floor(Math.random()*filteredPool.length)] ?? 0;
  const cfg = ENEMY_TYPES[idx];
  const isBoss = cfg.shape === 'boss';

  enemies.push({
    x, y: -45,
    config: cfg,
    vx: (Math.random()-0.5) * (isBoss ? 1.5 : 2.5),
    vy: enemySpeed * (isBoss ? 0.5 : 1),
    rot: 0, rotSpeed: (Math.random()-0.5) * (isBoss ? 0.02 : 0.05),
    hp: cfg.hp, maxHp: cfg.hp,
    shootTimer: Math.random()*60
  });
}

function collide(a, b, ra=16, rb=16) {
  const dx = a.x - b.x, dy = a.y - b.y;
  return Math.sqrt(dx*dx + dy*dy) < ra + rb;
}

// ── Main Loop ──
function gameLoop() {
  requestAnimationFrame(gameLoop);
  drawBackground();
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  if (state !== 'playing') return;

  frame++;

  // Level up
  const newLevel = 1 + Math.floor(score / 350);
  if (newLevel > level) {
    level = newLevel;
    enemySpeed = 1.7 + (level-1)*0.35;
    enemyInterval = Math.max(32, 85 - (level-1)*8);
    bCooldown = Math.max(7, 16 - (level-1)*2);
    showAnnounce(`⛓️ BLOCK ${level} — NETWORK UPGRADED`);
  }

  // Announce fade
  if (announceTimer > 0) {
    announceTimer--;
    if (announceTimer < 30) {
      document.getElementById('level-announce').style.opacity = announceTimer/30;
    }
    if (announceTimer === 0)
      document.getElementById('level-announce').style.opacity = 0;
  }

  // Spawn
  if (frame % enemyInterval === 0) spawnEnemy();

  // Move player
  if ((keys['ArrowLeft'] || keys['a']) && player)
    player.x = Math.max(PW/2, player.x - player.speed);
  if ((keys['ArrowRight'] || keys['d']) && player)
    player.x = Math.min(canvas.width-PW/2, player.x + player.speed);

  // Shoot
  shootTimer++;
  if ((keys[' '] || keys['ArrowUp'] || autoFire) && shootTimer >= bCooldown) {
    playerShoot(); shootTimer = 0;
  }

  // Invincibility
  if (player.invincible > 0) player.invincible--;
  if (player.shield) { player.shieldTimer--; if (player.shieldTimer <= 0) player.shield = false; }

  // Move bullets
  bullets = bullets.filter(b => b.y > -30);
  bullets.forEach(b => b.y += b.vy);

  // Move enemy bullets
  enemyBullets = enemyBullets.filter(b => b.y < canvas.height + 20);
  enemyBullets.forEach(b => { b.x += b.vx; b.y += b.vy; });

  // Move enemies
  enemies.forEach(e => {
    e.x += e.vx; e.y += e.vy;
    e.rot += e.rotSpeed;
    if (e.x < 25 || e.x > canvas.width-25) e.vx *= -1;

    e.shootTimer++;
    const sRate = e.config.shape === 'boss' ? 40 : Math.max(28, 75 - level*5);
    if (e.shootTimer > sRate) {
      e.shootTimer = 0;
      const ang = Math.atan2(player.y - e.y, player.x - e.x);
      const spd = e.config.shape === 'boss' ? 4.5 : 3;
      enemyBullets.push({ x:e.x, y:e.y, vx:Math.cos(ang)*spd, vy:Math.sin(ang)*spd, color:e.config.color });
      if (e.config.shape === 'boss') {
        // Spread shot
        [-0.4, 0.4].forEach(offset => {
          enemyBullets.push({ x:e.x, y:e.y, vx:Math.cos(ang+offset)*3.5, vy:Math.sin(ang+offset)*3.5, color:'#ff6060' });
        });
      }
    }
  });

  // Bullet–enemy collision
  bullets.forEach(b => {
    enemies.forEach(e => {
      const hr = e.config.shape==='boss' ? 26 : 18;
      if (collide(b, e, 5, hr)) {
        b._dead = true;
        e.hp--;
        spawnParticles(b.x, b.y, e.config.color, 5);
        if (e.hp <= 0) {
          score += e.config.score * level;
          spawnParticles(e.x, e.y, e.config.color, e.config.shape==='boss' ? 28 : 14, e.config.shape==='boss');
          spawnTokens(e.x, e.y);
          spawnPowerup(e.x, e.y);
          e._dead = true;
        }
      }
    });
  });
  bullets = bullets.filter(b => !b._dead);
  enemies = enemies.filter(e => !e._dead && e.y < canvas.height + 80);

  // Enemy bullet–player
  if (player.invincible === 0) {
    enemyBullets.forEach(b => {
      if (collide(b, player, 5, 14)) {
        b._dead = true;
        if (player.shield) {
          spawnParticles(player.x, player.y, '#06b6d4', 8);
        } else {
          lives--;
          player.invincible = 80;
          spawnParticles(player.x, player.y, '#ef4444', 20, true);
          if (lives <= 0) { endGame(); return; }
        }
      }
    });
  }
  enemyBullets = enemyBullets.filter(b => !b._dead);

  // Enemy–player
  enemies.forEach(e => {
    if (player.invincible === 0 && collide(e, player, e.config.shape==='boss'?26:16, 14)) {
      e._dead = true;
      if (!player.shield) {
        lives--;
        player.invincible = 80;
        spawnParticles(player.x, player.y, '#ef4444', 20, true);
        if (lives <= 0) { endGame(); return; }
      }
    }
  });
  enemies = enemies.filter(e => !e._dead);

  // Powerups
  powerups.forEach(p => {
    p.y += p.vy; p.anim += 0.07; p.life -= p.decay;
    if (collide(p, player, 14, 20)) {
      p._dead = true;
      if (p.type.id === 'shield') { player.shield = true; player.shieldTimer = 320; }
      else if (p.type.id === 'rapid') { bCooldown = Math.max(4, bCooldown-3); }
      else if (p.type.id === 'bomb') {
        enemies.forEach(e => {
          score += e.config.score * level;
          spawnParticles(e.x, e.y, e.config.color, 12);
          spawnTokens(e.x, e.y);
        });
        enemies = [];
        showAnnounce('💥 DAO BOMB — ALL THREATS CLEARED');
      }
    }
  });
  powerups = powerups.filter(p => !p._dead && p.life > 0 && p.y < canvas.height + 30);

  // Tokens (floating $DLI)
  tokens.forEach(t => {
    t.x += t.vx; t.y += t.vy; t.life -= t.decay; t.anim += 0.1;
    t.vy += 0.03;
  });
  tokens = tokens.filter(t => t.life > 0);

  // Particles
  particles.forEach(p => { p.x+=p.vx; p.y+=p.vy; p.vy+=0.04; p.life-=p.decay; });
  particles = particles.filter(p => p.life > 0);

  // ── Draw ──

  // Powerups
  powerups.forEach(p => {
    ctx.save();
    ctx.translate(p.x, p.y);
    ctx.globalAlpha = p.life;
    ctx.shadowBlur = 14; ctx.shadowColor = p.type.color;
    ctx.beginPath(); ctx.arc(0, 0, 20, 0, Math.PI*2);
    ctx.fillStyle = `${p.type.color}22`; ctx.fill();
    ctx.strokeStyle = p.type.color; ctx.lineWidth = 1.5; ctx.stroke();
    ctx.font = '18px serif'; ctx.textAlign='center'; ctx.textBaseline='middle';
    ctx.fillText(p.type.icon, 0, 0);
    ctx.font = 'bold 6px Orbitron, monospace';
    ctx.fillStyle = p.type.color; ctx.textBaseline='bottom';
    ctx.fillText(p.type.label, 0, 30);
    ctx.restore();
  });

  // Floating $DLI tokens
  tokens.forEach(t => {
    ctx.save();
    ctx.translate(t.x, t.y);
    ctx.globalAlpha = t.life;
    ctx.shadowBlur = 8; ctx.shadowColor = '#f59e0b';
    ctx.font = 'bold 8px Orbitron, monospace';
    ctx.fillStyle = '#f59e0b'; ctx.textAlign='center'; ctx.textBaseline='middle';
    ctx.fillText('$DLI', 0, 0);
    ctx.restore();
  });

  // Enemies
  enemies.forEach(drawEnemy);

  // Player bullets
  bullets.forEach(b => {
    ctx.save();
    const isMain = b.type === 'normal';
    ctx.shadowBlur = isMain ? 12 : 8;
    ctx.shadowColor = isMain ? '#06b6d4' : '#a78bfa';
    const bg = ctx.createLinearGradient(b.x, b.y, b.x, b.y+b.h);
    bg.addColorStop(0, '#ffffff');
    bg.addColorStop(0.4, isMain ? '#06b6d4' : '#a78bfa');
    bg.addColorStop(1, isMain ? '#0e7490' : '#7c3aed');
    ctx.fillStyle = bg;
    ctx.beginPath(); ctx.roundRect(b.x-b.w/2, b.y, b.w, b.h, 2); ctx.fill();
    ctx.restore();
  });

  // Enemy bullets
  enemyBullets.forEach(b => {
    ctx.save();
    ctx.shadowBlur = 10; ctx.shadowColor = b.color;
    ctx.beginPath(); ctx.arc(b.x, b.y, 5, 0, Math.PI*2);
    ctx.fillStyle = b.color; ctx.fill();
    ctx.restore();
  });

  // Particles
  particles.forEach(p => {
    ctx.save(); ctx.globalAlpha = p.life;
    ctx.shadowBlur = 6; ctx.shadowColor = p.color;
    ctx.beginPath(); ctx.arc(p.x, p.y, p.r*p.life, 0, Math.PI*2);
    ctx.fillStyle = p.color; ctx.fill();
    ctx.restore();
  });

  // Player
  if (player) drawPlayer(player);

  updateHUD();
}

function startGame() {
  document.getElementById('start-screen').style.display = 'none';
  document.getElementById('gameover-screen').style.display = 'none';
  initGame();
  state = 'playing';
  showAnnounce('🚀 DLICOM DEFENDER — PROTECT THE CHAIN');
}

function endGame() {
  state = 'dead';
  if (score > hiScore) hiScore = score;
  document.getElementById('final-score').textContent = score.toLocaleString();
  const msg = score >= 2000 ? `🔥 Chain Master! Rekor: ${hiScore.toLocaleString()} $DLI` :
               score >= 800  ? `💪 Defender Handal! Rekor: ${hiScore.toLocaleString()} $DLI` :
               `Blockchain perlu perlindungan lebih! Terbaik: ${hiScore.toLocaleString()} $DLI`;
  document.getElementById('final-msg').textContent = msg;
  document.getElementById('gameover-screen').style.display = 'flex';

  if (tg) {
    try { tg.sendData?.(JSON.stringify({ score, level, hiScore, game:'dlicom-defender' })); } catch(e) {}
  }
}
